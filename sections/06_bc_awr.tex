\section{Behavior Cloning and AWR}

\begin{frame}{Behavior Cloning}
  \Large
  \[
    \mathcal L_{\mathrm{BC}}(\theta)
    =
    -\mathbb E_{(s,a)\sim \dataset}
    \left[\log \pi_\theta(a \mid s)\right]
  \]

  \vspace{0.4cm}
  \normalsize
  Learn to imitate the actions in the dataset.

  \vspace{0.3cm}
  Benefits: stable, simple, scalable, no explicit value learning.
  \note{Behavior cloning is just supervised learning on logged actions. It is the baseline every offline RL method should beat.}
\end{frame}

\begin{frame}{BC Clones All Actions Equally}
  \centering
  \begin{tikzpicture}[x=1.2cm,y=0.8cm]
    \node[left] at (0,3) {action A};
    \node[left] at (0,2) {action B};
    \node[left] at (0,1) {action C};
    \foreach \y in {1,2,3} {
      \draw[fill=Panel] (0,\y-0.25) rectangle (2.4,\y+0.25);
    }
    \node[right] at (2.5,3) {low return};
    \node[right] at (2.5,2) {medium return};
    \node[right] at (2.5,1) {high return};
  \end{tikzpicture}

  \vspace{0.4cm}
  BC does not know which actions were good. It clones good, mediocre, and bad actions alike.
  \note{This sets up AWR. BC is stable because it stays in the data, but it does not perform value-aware improvement.}
\end{frame}

\begin{frame}{Advantage-Weighted Regression}
  \Large
  \[
    \mathcal L_{\mathrm{AWR}}(\theta)
    =
    -\mathbb E_{(s,a)\sim \dataset}
    \left[w(s,a)\log \pi_\theta(a \mid s)\right]
  \]

  \[
    w(s,a) \propto \exp\left(\frac{A(s,a)}{\lambda}\right),
    \qquad
    A(s,a)=Q(s,a)-V(s)
  \]
  \note{Present AWR as a minimal conceptual step: still supervised learning, but weighted by long-term value. Cite Peng et al. verbally here.}
\end{frame}

\begin{frame}{BC vs. AWR Weighting}
  \centering
  \begin{tikzpicture}[x=1.15cm,y=0.85cm]
    \node at (1.2,4) {BC};
    \node at (5.4,4) {AWR};
    \foreach \y/\lab/\ret in {3/A/low,2/B/medium,1/C/high} {
      \node[left] at (0,\y) {\lab};
      \draw[fill=Panel] (0.4,\y-0.25) rectangle (2.0,\y+0.25);
      \node[left] at (4.0,\y) {\lab};
    }
    \draw[fill=WarmGold!35] (4.4,2.75) rectangle (5.1,3.25);
    \draw[fill=WarmGold!55] (4.4,1.75) rectangle (5.8,2.25);
    \draw[fill=SignalGreen!55] (4.4,0.75) rectangle (7.1,1.25);
    \node[right] at (7.2,1) {cloned most};
  \end{tikzpicture}

  \vspace{0.3cm}
  High-advantage actions are cloned more strongly; low-advantage actions are downweighted.
  \note{Use this example to make AWR intuitive. If the logs contain mixed-quality decisions, AWR preferentially imitates the better parts.}
\end{frame}

\begin{frame}{AWR Is the First RL Step}
  \Large
  AWR is the smallest conceptual step from imitation learning to reinforcement learning.

  \vspace{0.5cm}
  \normalsize
  It still looks like supervised learning, but the labels are now weighted by long-term value.

  \vspace{0.4cm}
  \productbox{If logs contain good and bad decisions, AWR gives a way to imitate the better ones more.}
  \note{Transition to IQL: AWR needs advantages. IQL is a modern way to learn them in offline data without out-of-distribution maximization.}
\end{frame}
