\section{Idea 3: In-Sample Value Learning}

\begin{frame}{A Different Idea: Never Evaluate Unseen Actions}
  CQL keeps the maximization, but makes unsupported actions pessimistic.

  \vspace{0.25cm}
  Can we avoid the maximization entirely?

  \vspace{0.15cm}
  Standard Q-learning uses the target:
  \[
    Q(s,a)
    \leftarrow
    r+\gamma\,\textcolor{red}{\max_{a'} Q(s',a')}.
  \]

  \vspace{0.25cm}
  The problem is that the \textcolor{red}{maximization} evaluates actions that never appeared in the dataset, creating extrapolation errors.

  \vspace{0.25cm}
  \begin{block}{Main takeaway}
    IQL's key idea: never evaluate actions that are not in the dataset.
  \end{block}
  \note{Start from the question left open by CQL. The key issue is the max over actions in the Bellman target. IQL does not reject value learning; it avoids evaluating unseen actions.}
\end{frame}

\begin{frame}{Learn the Best Supported Value}
  Instead of computing $\max_a Q(s,a)$, IQL learns a state value:
  \[
    V(s)
    \approx
    \text{upper expectile of } Q(s,a)
    \quad \text{for logged actions } a.
  \]

  \vspace{0.1cm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.44\textwidth}
      \begin{itemize}
        \item mean: average action
        \item maximum: possibly unsupported action
        \item upper expectile: high but supported action
      \end{itemize}
    \end{column}
    \begin{column}{0.54\textwidth}
      \centering
      \begin{tikzpicture}[>=Latex,x=0.78cm,y=0.58cm]
        \draw[->,thick] (0,0) -- (7.7,0) node[right] {action};
        \foreach \x/\h in {0.8/0.9,1.5/1.8,2.3/1.3,3.1/2.5,4.0/2.1,4.8/1.5} {
          \draw[fill=Panel] (\x-0.11,0) rectangle (\x+0.11,\h);
        }
        \node[above] at (2.8,2.85) {logged actions};
        \draw[SignalGreen,ultra thick] (0.55,2.25) -- (5.05,2.25);
        \node[right] at (5.1,2.25) {$V(s)$};
        \node[right,text=SignalGreen] at (5.1,1.75) {high supported value};
        \draw[WarmGold,dashed,thick] (6.45,0) -- (6.45,3.05);
        \node[above] at (6.45,3.05) {unsupported};
      \end{tikzpicture}
    \end{column}
  \end{columns}

  \vspace{0.15cm}
  Expectile regression is the tool used to estimate this high supported value.

  \vspace{0.05cm}
  Then learn $Q$ with:
  \[
    Q(s,a) \leftarrow r+\gamma V(s').
  \]

  \vspace{-0.05cm}
  Notice: $\max_a Q(s,a)$ has disappeared.
  \note{Do not derive expectile regression. Present it as the implementation detail that estimates a high value among logged actions. The mathematical point is that Q backs up through V, not a max over all actions.}
\end{frame}

\begin{frame}{Policy Extraction: Reuse AWR}
  Once $V(s)$ and $Q(s,a)$ have been learned, the policy update is familiar.

  \vspace{0.1cm}
  \begin{center}
    \begin{tikzpicture}[node distance=0.18cm,>=Latex]
      \tikzset{
        flowstep/.style={
          draw,
          rounded corners,
          fill=Panel,
          minimum width=5.0cm,
          minimum height=0.5cm,
          align=center
        }
      }
      \node[flowstep] (data) {logged dataset};
      \node[flowstep,below=of data] (v) {learn $V$ \quad {\small high supported value}};
      \node[flowstep,below=of v] (q) {learn $Q$ \quad {\small $Q(s,a)\leftarrow r+\gamma V(s')$}};
      \node[flowstep,below=of q] (a) {compute $A=Q-V$};
      \node[flowstep,below=of a] (awr) {run AWR \quad {\small $w=e^{\beta A}$}};
      \draw[->,thick] (data) -- (v);
      \draw[->,thick] (v) -- (q);
      \draw[->,thick] (q) -- (a);
      \draw[->,thick] (a) -- (awr);
    \end{tikzpicture}
  \end{center}

  \vspace{0.15cm}
  \begin{block}{Main takeaway}
    IQL changes value estimation, not policy optimization:
    \(\boxed{V \rightarrow Q \rightarrow A \rightarrow \pi}\)
  \end{block}
  \note{Close the loop back to AWR. IQL is not a completely different policy update; it is better offline value estimation plugged into the same advantage-weighted policy extraction.}
\end{frame}
