\section{Idea 2: Conservative Value Learning}

\begin{frame}{Problem: Offline Values Can Be Overconfident}
  To compute useful weights, we need value estimates.

  \vspace{0.35cm}
  But value estimation is harder offline:
  \begin{itemize}
    \item online learning can test actions and correct bad guesses,
    \item offline learning cannot collect new feedback,
    \item unsupported actions may receive overly optimistic predicted values.
  \end{itemize}

  \vspace{0.35cm}
  \begin{block}{Question}
    How can we learn values without trusting predictions where the data is weak?
  \end{block}
  \note{This is the only place to briefly reintroduce why offline value learning is hard. Keep it short.}
\end{frame}

\begin{frame}{Core Idea: Conservative Value Learning}
  \centering
  \begin{tikzpicture}[>=Latex,x=0.82cm,y=0.52cm]
    \draw[->,thick] (0,0) -- (10,0) node[right] {action};
    \foreach \x/\h in {1.4/1.0,2.2/1.7,3.0/1.3,3.9/2.0,4.8/1.5} {
      \draw[fill=Panel] (\x-0.12,0) rectangle (\x+0.12,\h);
    }
    \node[above] at (3.1,2.8) {training data};
    \draw[SignalGreen,ultra thick] (1.0,1.9) -- (5.2,1.9);
    \node[right] at (5.25,1.9) {high $Q$};
    \draw[WarmGold,dashed,ultra thick] (7.6,0) -- (7.6,2.8);
    \node[above] at (7.6,2.8) {unsupported};
    \draw[WarmGold,ultra thick] (7.0,0.45) -- (8.2,0.45);
    \node[right] at (8.25,0.85) {low $Q$};
  \end{tikzpicture}

  \vspace{0.15cm}
  \[
    \mathcal L_{\rm CQL}
    =
    \underbrace{\mathcal L_{\rm Bellman}}_{\text{fit observed transitions}}
    +
    \alpha
    \left(
      \underbrace{\mathbb E_{a\sim\mu} Q(s,a)}_{\text{push down candidate actions}}
      -
      \underbrace{\mathbb E_{(s,a)\sim\mathcal D} Q(s,a)}_{\text{keep logged actions high}}
    \right).
  \]

  \vspace{0.15cm}
  \begin{block}{Main takeaway}
    \textbf{CQL changes only the value-learning objective:} fit the data, but penalize high $Q$-values for unsupported actions.
  \end{block}
  \note{This slide combines the CQL picture with the schematic objective. Mu represents candidate actions used to probe unsupported or weakly supported regions. The audience should remember: Bellman loss plus conservative penalty.}
\end{frame}
